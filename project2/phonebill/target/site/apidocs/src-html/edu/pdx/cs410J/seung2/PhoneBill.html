001package edu.pdx.cs410J.seung2;
002
003import edu.pdx.cs410J.AbstractPhoneBill;
004
005import java.util.ArrayList;
006import java.util.Collection;
007import java.util.Iterator;
008
009public class PhoneBill extends AbstractPhoneBill<PhoneCall> {
010    private String customer;
011    private Collection<PhoneCall> calls = new ArrayList<>();
012
013    public void setCustomer(String[] args){
014        this.customer = args[0];
015
016        for(int i = 1; i < args.length; i++){
017            this.customer += " " + args[i];
018        }
019    }
020
021    @Override
022    public String getCustomer() {
023        return this.customer;
024    }
025
026    @Override
027    public void addPhoneCall(PhoneCall phoneCall) {
028        boolean checker = this.calls.add(phoneCall);
029
030        if(!checker){
031            System.err.println("Adding phonecall to the phonebill was unsuccessful");
032            System.exit(1);
033        }
034    }
035
036    @Override
037    public Collection<PhoneCall> getPhoneCalls() {
038        return this.calls;
039    }
040}
