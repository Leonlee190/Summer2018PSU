001package edu.pdx.cs410J.seung2;
002
003import edu.pdx.cs410J.AbstractPhoneCall;
004
005public class PhoneCall extends AbstractPhoneCall {
006  private String caller = "not implemented";
007  private String callee = "not implemented";
008  private String startTime = "not implemented";
009  private String endTime = "not implemented";
010
011  public void initCall(String[] Caller, String[] Callee, String[] sDate, String[] eDate, String[] sTime, String[] eTime){
012    this.caller = Caller[0] + "-" + Caller[1] + "-" + Caller[2];
013    this.callee = Callee[0] + "-" + Callee[1] + "-" + Callee[2];
014    this.startTime = sDate[0] + "/" + sDate[1] + "/" + sDate[2] + " " + sTime[0] + ":" + sTime[1];
015    this.endTime = eDate[0] + "/" + eDate[1] + "/" + eDate[2] + " " + eTime[0] + ":" + eTime[1];
016  }
017
018  @Override
019  public String getCaller() {
020    return this.caller;
021  }
022
023  @Override
024  public String getCallee() {
025    return this.callee;
026  }
027
028  @Override
029  public String getStartTimeString() {
030    return this.startTime;
031  }
032
033  @Override
034  public String getEndTimeString() {
035    return this.endTime;
036  }
037}
