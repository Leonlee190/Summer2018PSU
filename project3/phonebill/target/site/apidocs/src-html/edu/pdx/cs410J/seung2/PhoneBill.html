001package edu.pdx.cs410J.seung2;
002
003import edu.pdx.cs410J.AbstractPhoneBill;
004
005import java.util.ArrayList;
006import java.util.Collection;
007import java.util.Iterator;
008
009/**
010 * PhoneBill class that stores and returns customer name and collection of PhoneCall class
011 */
012public class PhoneBill extends AbstractPhoneBill<PhoneCall> {
013    private String customer;                                        // Stores customer's name
014    private Collection<PhoneCall> calls = new ArrayList<PhoneCall>();        // Array list of PhoneCall classes
015
016    /**
017     * Parses and sets the customer's name
018     *
019     * @param args
020     *         Array of String that was parsed from command line argument
021     */
022    public void setCustomer(String[] args){
023        // Get the first name
024        this.customer = args[0];
025
026        // Get rest of the name
027        for(int i = 1; i < args.length; i++){
028            this.customer += " " + args[i];
029        }
030    }
031
032    /**
033     * Returns the customer's name
034     *
035     * @return
036     *          Returns the customer's name in String form
037     */
038    @Override
039    public String getCustomer() {
040        return this.customer;
041    }
042
043    /**
044     * Add single PhoneCall class object into the collection
045     *
046     * @param phoneCall
047     *         PhoneCall class object initialized in the Project1's main
048     */
049    @Override
050    public void addPhoneCall(PhoneCall phoneCall) {
051        // Check if the adding is correctly done
052        boolean checker = this.calls.add(phoneCall);
053
054        // If not added correctly then error
055        if(!checker){
056            System.err.println("Adding phonecall to the phonebill was unsuccessful");
057            System.exit(1);
058        }
059    }
060
061    /**
062     * Returns the PhoneCall class collection
063     *
064      * @return
065     *           PhoneCall collection stored in the PhoneBill class
066     */
067    @Override
068    public Collection<PhoneCall> getPhoneCalls() {
069        return this.calls;
070    }
071}
