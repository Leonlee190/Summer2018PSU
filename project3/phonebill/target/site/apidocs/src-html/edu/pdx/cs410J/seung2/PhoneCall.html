001package edu.pdx.cs410J.seung2;
002
003import edu.pdx.cs410J.AbstractPhoneCall;
004
005/**
006 * PhoneCall class that stores and initialize customer's call information
007 */
008public class PhoneCall extends AbstractPhoneCall {
009  private String caller = "not implemented";          // Caller's number
010  private String callee = "not implemented";          // Callee's number
011  private String startTime = "not implemented";       // Phone call's starting date and time
012  private String endTime = "not implemented";         // Phone call's ending date and time
013
014  /**
015   * Initializes the PhoneCall's field with checked but fully parsed fields
016   *
017   * @param Caller
018   *         String array of caller's number in three sections
019   * @param Callee
020   *         String array of callee's number in three sections
021   * @param sDate
022   *         String array of starting date which has month, date, and year value
023   * @param eDate
024   *         String array of ending date which has month, date, and year value
025   * @param sTime
026   *         String array of starting time which has hour and min
027   * @param eTime
028   *         String array of ending time which has hour and min
029   */
030  public void initCall(String[] Caller, String[] Callee, String[] sDate, String[] eDate, String[] sTime, String[] eTime){
031    // Initialize caller and callee's phone number with correct format
032    this.caller = Caller[0] + "-" + Caller[1] + "-" + Caller[2];
033    this.callee = Callee[0] + "-" + Callee[1] + "-" + Callee[2];
034
035    // Initialize starting and ending date and time with correct format
036    this.startTime = sDate[0] + "/" + sDate[1] + "/" + sDate[2] + " " + sTime[0] + ":" + sTime[1];
037    this.endTime = eDate[0] + "/" + eDate[1] + "/" + eDate[2] + " " + eTime[0] + ":" + eTime[1];
038  }
039
040  /**
041   * Returns the caller's number
042   *
043   * @return
044   *          Returns the caller's number stored in PhoneCall class
045   */
046  @Override
047  public String getCaller() {
048    return this.caller;
049  }
050
051  /**
052   * Returns the callee's number
053   *
054   * @return
055   *          Returns the callee's number stored in PhoneCall class
056   */
057  @Override
058  public String getCallee() {
059    return this.callee;
060  }
061
062  /**
063   * Returns the starting date and time
064   *
065   * @return
066   *          Returns the starting date and time stored in PhoneCall class
067   */
068  @Override
069  public String getStartTimeString() {
070    return this.startTime;
071  }
072
073  /**
074   * Returns ending date and time
075   *
076   * @return
077   *          Returns the ending date and time stored in PhoneCall class
078   */
079  @Override
080  public String getEndTimeString() {
081    return this.endTime;
082  }
083}
