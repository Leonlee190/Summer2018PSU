026  public static void main(String[] args) {
027    PhoneCall call = new PhoneCall();  // Refer to one of Dave's classes so that we can be sure it is on the classpath
028    PhoneBill bill = new PhoneBill();  // PhoneBill class to contain customer name and PhoneCall data
029    TextDumper dumper = new TextDumper(); // TextDumper class for -textFile option
030    TextParser parser = new TextParser(); // TextParser class for -textFile option
031
032    // If no command line argument
033    if(args.length < 1){
034      System.err.println("Missing command line arguments");
035      System.exit(1);
036    }
037
038    // Only use -textFile option if it's "-textFile" spelling and with 8 arguments
039    if(args[0].equals("-textFile") && args.length == 9){
040      // Check if the file name is correct
041      checkFilename(args[1]);
042
043      // Retrieve file name from the argument
044      String fileName = args[1];
045
046      // Initialize the PhoneBill and PhoneCall from command line argument
047      parseInput(call, bill, args, 2);
048      addCalls(bill, call);
049
050      // Within try and catch for IOException
051      try {
052        // Open up the file
053        File fl = new File(fileName);
054
055        // Create new file
056        boolean checker = fl.createNewFile();
057
058        // If file didn't exist previously and has created new file
059        if(checker){
060          try {
061            // Pass in the file name to the TextDumper
062            dumper.setFl(fileName);
063
064            // Dump all the information onto the file
065            dumper.dump(bill);
066          }catch(IOException e){
067            // Catch the IOException thrown from the dump method
068            System.err.println("Creating new file and writing has failed");
069            System.exit(1);
070          }
071          System.exit(0);
072        }
073        // If the text file already exist with that name
074        else{
075          // If the file exists but empty then error
076          if(fl.length() == 0){
077            System.err.println("Nothing in the file. Invalid file.");
078            System.exit(1);
079          }
080
081          try {
082            // Set up the parser with the file name
083            parser.setFl(fileName);
084
085            // Retrieve the AbstractPhoneBill returned by file parsing
086            AbstractPhoneBill getBill = parser.parse();
087
088            // If text file's customer and command line's customer does not match then error
089            if(!getBill.getCustomer().equals(bill.getCustomer())){
090              System.err.println("Input Phone Bill customer name does not match the file's customer name");
091              System.exit(1);
092            }
093
094            // Check if the command line's PhoneCall already exist
095            checkCallDupli(getBill.getPhoneCalls(), call);
096
097            // Add the command line's PhoneCall to the retrieved PhoneBill
098            getBill.addPhoneCall(call);
099
100            // Catch IOException because it's dump not parse
101            try {
102              // Set up the dumper with the file name and dump
103              dumper.setFl(fileName);
104              dumper.dump(getBill);
105            }catch(IOException e){
106              System.err.println("Writing onto file failed after reading from " + fileName);
107              System.exit(1);
108            }
109          }catch(ParserException e){          // If Parsing was invalid
110            System.err.println("Couldn't parse the " + fileName + " file");
111            System.exit(1);
112          }
113          System.exit(0);
114        }
115      }catch(IOException e){      // If initial file I/O was a failure
116        System.err.println("File I/O has failed");
117        System.exit(1);
118      }
119
120      // Exit if all done
121      System.exit(0);
122    }
123
124    // If only print option is called without any initialization
125    if(args[0].equals("-print")  && args.length == 1){
126      System.err.println("No phone call information to print");
127      System.exit(1);
128    }
129
130    // If -print option with data is read
131    if(args[0].equals("-print")  && args.length == 8){
132      parseInput(call, bill, args, 1);
133      addCalls(bill, call);
134      System.out.println(call.toString());
135      System.exit(0);
136    }
137    // Only -readme option is called
138    else if(args[0].equals("-README") && args.length == 1){
139      System.out.println(README);
140      System.exit(0);
141    }
142    // Just call data has been inputted
143    else if(args.length == 7){
144      parseInput(call, bill, args, 0);
145      addCalls(bill, call);
146      System.out.println(call.toString());
147      System.exit(0);
148    }
149    // Everything else returns error
150    else{
151      System.err.println("Missing command line arguments");
152      System.exit(1);
153    }
154
155    // In non of the options were followed then exit with invalid
156    System.exit(1);
157  }

161   *
162   * @param input
163   *         String to check
164   * @return boolean
165   *          if the String is int or not with boolean
166   */
167  public static boolean isInt(String input){
168    boolean isIt;
169
170    // Try parsing String to Int, if can't then return false
171    try{
172      Integer.parseInt(input);
173      isIt = true;
174    }catch(NumberFormatException e){
175      isIt = false;
176    }
177
178    return isIt;
179  }

202  /**
203   *  Test date format
204   *  Valid : MM/DD/YYYY
205   *      - Must be using "/" to separate the values
206   *      - Month and Day needs to be 1 or 2 digits
207   *      - Year needs to be 4 digits only
208   *      - Month needs to be between 1 and 12
209   *      - Day needs to be between 1 and 31
210   *      - Should only have month, day, and year and no more
211   *
212   * @param args
213   *         String array which was split from command line
214   * @param arg
215   *         Corresponding command line argument
216   * @param name
217   *         Name of the format
218   */
219  public static void checkDate(String[] args, String arg, String name) {
220    checkDateFormat(arg, name);
221    // If there aren't Month, Day, and Year or more than that then error
222    if(args.length != 3){
223      System.err.println("Incorrect " + name + " date value!");
224      System.exit(1);
225    }
226
227    // If Month and Day has more than 2 digits then error
228    if((args[0].length() > 2) || (args[1].length() > 2)){
229      System.err.println("Incorrect " + name + " time value!");
230      System.exit(1);
231    }
232
233    // Change Month and Day string to int
234    int month = Integer.parseInt(args[0]);
235    int day = Integer.parseInt(args[1]);
236
237    // If the month value is outside the bound then error
238    if (month < 1 || month > 12) {
239      System.err.println("Incorrect " + name + " month value!");
240      System.exit(1);
241    }
242
243    // If the day value is outside the bound then error
244    if (day < 1 || day > 31) {
245      System.err.println("Incorrect " + name + " date value!");
246      System.exit(1);
247    }
248
249    // If year isn't in 4 digit then error
250    if (args[2].length() != 4) {
251      System.err.println("Incorrect " + name + " year value!");
252      System.exit(1);
253    }
254  }

256  /**
257   *  Check time format
258   *
259   *  Valid : HH:MM
260   *      - Must be using ":" to separate the values
261   *      - Hour and Min should be the only input
262   *      - Hour and Min is 1 or 2 digits
263   *      - Hour needs to be between 0 and 23
264   *      - Min needs to be between 0 and 59
265   *
266   * @param args
267   *         String array split from the command line
268   * @param arg
269   *         Corresponding command line argument
270   * @param name
271   *         Name of the format
272   */
273  public static void checkTime(String[] args, String arg, String name){
274    checkTimeFormat(arg, name);
275
276    // If there is less or more than 2 inputs (Hour and Min) then error
277    if(args.length != 2){
278      System.err.println("Incorrect " + name + " time value!");
279      System.exit(1);
280    }
281
282    // If hour and min is bigger than 2 digits then error
283    if((args[0].length() > 2) || (args[1].length() > 2)){
284      System.err.println("Incorrect " + name + " time value!");
285      System.exit(1);
286    }
287
288    // String to Int for hour and min value
289    int hour = Integer.parseInt(args[0]);
290    int min = Integer.parseInt(args[1]);
291
292    // If Hour is outside the bound then error
293    if(hour < 0 || hour > 24){
294      System.err.println("Incorrect " + name + " hour value!");
295      System.exit(1);
296    }
297
298    // If Min is outside the bound then error
299    if(min < 0 || min >= 60){
300      System.err.println("Incorrect " + name + " minute value!");
301      System.exit(1);
302    }
303  }

373  /**
374   * Parses the command line argument and initialize the PhoneCall field and add it to the PhoneBill field
375   *    Argument in order of
376   *        - Customer Name
377   *        - Caller Number
378   *        - Callee Number
379   *        - Starting Date
380   *        - Starting Time
381   *        - Ending Date
382   *        - Ending Time
383   *
384   * @param calling
385   *         PhoneCall field for the command line input
386   * @param billing
387   *         PhoneBill field for Customer name and PhoneCall field
388   * @param args
389   *         Command line argument
390   * @param i
391   *         Option buffer
392   */
393  public static void parseInput(PhoneCall calling, PhoneBill billing, String[] args, int i){
394    // Split with space and set PhoneBill's customer name
395    String[] name = args[i].split(" ");
396    billing.setCustomer(name);
397
398    // Split with "-" and check for caller number validity
399    String[] caller = args[i+1].split("-");
400    checkInt(caller, "caller");
401    checkNumberFormat(args[i+1], "caller");
402
403    // Split with "-" and check for callee number validity
404    String[] callee = args[i+2].split("-");
405    checkInt(callee, "callee");
406    checkNumberFormat(args[i+2], "callee");
407
408    // Split with "/" and check for starting date validity
409    String[] start = args[i+3].split("/");
410    checkInt(start, "starting date");
411    checkDate(start, args[i+3], "starting date's");
412
413    // Split with ":" and check for starting time validity
414    String[] startTime = args[i+4].split(":");
415    checkInt(startTime, "starting time");
416    checkTime(startTime, args[i+4], "starting time");
417
418    // Split with "/" and check for ending date validity
419    String[] end = args[i+5].split("/");
420    checkInt(end, "ending date");
421    checkDate(end, args[i+5], "ending date's");
422
423    // Split with ":" and check for ending time validity
424    String[] endTime = args[i+6].split(":");
425    checkInt(endTime, "ending time");
426    checkTime(endTime, args[i+6], "ending time");
427
428    // Initialize PhoneCall class with verified command line input
429    calling.initCall(caller, callee, start, end, startTime, endTime);
430  }

493  /**
494   * Check if the command line fed PhoneCall information already exists in the PhoneBill class
495   *
496   * @param calls
497   *         Collection of PhoneCall in PhoneBill
498   *
499   * @param calling
500   *         PhoneCall initialized from command line argument
501   */
502  public static void checkCallDupli(Collection<PhoneCall> calls, PhoneCall calling){
503    Iterator<PhoneCall> iter = calls.iterator();      // Set up Iterator to go through all the PhoneCalls
504
505    // Keep going until no object is left
506    while(iter.hasNext()){
507      PhoneCall obj = iter.next();
508
509      // If current iteration of PhoneCall is exactly same as PhoneCall from command line, then send out error
510      if(obj.getCaller().equals(calling.getCaller()) && obj.getCallee().equals(calling.getCallee()) && obj.getStartTimeString().equals(calling.getStartTimeString())){
511        if(obj.getEndTimeString().equals(calling.getEndTimeString())){
512          System.err.println("Inputted Phone Call already exists");
513          System.exit(1);
514        }
515      }
516    }
517  }
