26    public static void main(String[] args) {
27      PhoneCall call = new PhoneCall();  // Refer to one of Dave's classes so that we can be sure it is on the classpath
28      PhoneBill bill = new PhoneBill();  // PhoneBill class to contain customer name and PhoneCall data
29      TextDumper dumper = new TextDumper(); // TextDumper class for -textFile option
30      TextParser parser = new TextParser(); // TextParser class for -textFile option
31  
32      // If no command line argument
33      if(args.length < 1){
34        System.err.println("Missing command line arguments");
35        System.exit(1);
36      }
37  
38      // Only use -textFile option if it's "-textFile" spelling and with 8 arguments
39      if(args[0].equals("-textFile") && args.length == 9){
40        // Check if the file name is correct
41        checkFilename(args[1]);
42  
43        // Retrieve file name from the argument
44        String fileName = args[1];
45  
46        // Initialize the PhoneBill and PhoneCall from command line argument
47        parseInput(call, bill, args, 2);
48        addCalls(bill, call);
49  
50        // Within try and catch for IOException
51        try {
52          // Open up the file
53          File fl = new File(fileName);
54  
55          // Create new file
56          boolean checker = fl.createNewFile();
57  
58          // If file didn't exist previously and has created new file
59          if(checker){
60            try {
61              // Pass in the file name to the TextDumper
62              dumper.setFl(fileName);
63  
64              // Dump all the information onto the file
65              dumper.dump(bill);
66            }catch(IOException e){
67              // Catch the IOException thrown from the dump method
68              System.err.println("Creating new file and writing has failed");
69              System.exit(1);
70            }
71            System.exit(0);
72          }
73          // If the text file already exist with that name
74          else{
75            // If the file exists but empty then error
76            if(fl.length() == 0){
77              System.err.println("Nothing in the file. Invalid file.");
78              System.exit(1);
79            }
80  
81            try {
82              // Set up the parser with the file name
83              parser.setFl(fileName);
84  
85              // Retrieve the AbstractPhoneBill returned by file parsing
86              AbstractPhoneBill getBill = parser.parse();
87  
88              // If text file's customer and command line's customer does not match then error
89              if(!getBill.getCustomer().equals(bill.getCustomer())){
90                System.err.println("Input Phone Bill customer name does not match the file's customer name");
91                System.exit(1);
92              }
93  
94              // Check if the command line's PhoneCall already exist
95              checkCallDupli(getBill.getPhoneCalls(), call);
96  
97              // Add the command line's PhoneCall to the retrieved PhoneBill
98              getBill.addPhoneCall(call);
99  
100             // Catch IOException because it's dump not parse
101             try {
102               // Set up the dumper with the file name and dump
103               dumper.setFl(fileName);
104               dumper.dump(getBill);
105             }catch(IOException e){
106               System.err.println("Writing onto file failed after reading from " + fileName);
107               System.exit(1);
108             }
109           }catch(ParserException e){          // If Parsing was invalid
110             System.err.println("Couldn't parse the " + fileName + " file");
111             System.exit(1);
112           }
113           System.exit(0);
114         }
115       }catch(IOException e){      // If initial file I/O was a failure
116         System.err.println("File I/O has failed");
117         System.exit(1);
118       }
119 
120       // Exit if all done
121       System.exit(0);
122     }
123 
124     // If only print option is called without any initialization
125     if(args[0].equals("-print")  && args.length == 1){
126       System.err.println("No phone call information to print");
127       System.exit(1);
128     }

206    *      - Month and Day needs to be 1 or 2 digits
207    *      - Year needs to be 4 digits only

221     // If there aren't Month, Day, and Year or more than that then error
222     if(args.length != 3){
223       System.err.println("Incorrect " + name + " date value!");
224       System.exit(1);
225     }
226 
227     // If Month and Day has more than 2 digits then error
228     if((args[0].length() > 2) || (args[1].length() > 2)){
229       System.err.println("Incorrect " + name + " time value!");
230       System.exit(1);
231     }
232 
233     // Change Month and Day string to int
234     int month = Integer.parseInt(args[0]);
235     int day = Integer.parseInt(args[1]);
236 
237     // If the month value is outside the bound then error
238     if (month < 1 || month > 12) {
239       System.err.println("Incorrect " + name + " month value!");
240       System.exit(1);
241     }
242 
243     // If the day value is outside the bound then error
244     if (day < 1 || day > 31) {
245       System.err.println("Incorrect " + name + " date value!");

247     }

373   /**
374    * Parses the command line argument and initialize the PhoneCall field and add it to the PhoneBill field
375    *    Argument in order of
376    *        - Customer Name
377    *        - Caller Number
378    *        - Callee Number
379    *        - Starting Date
380    *        - Starting Time
381    *        - Ending Date
382    *        - Ending Time
383    *
384    * @param calling
385    *         PhoneCall field for the command line input
386    * @param billing
387    *         PhoneBill field for Customer name and PhoneCall field
388    * @param args
389    *         Command line argument
390    * @param i
391    *         Option buffer
392    */
393   public static void parseInput(PhoneCall calling, PhoneBill billing, String[] args, int i){
394     // Split with space and set PhoneBill's customer name
395     String[] name = args[i].split(" ");
396     billing.setCustomer(name);
397 
398     // Split with "-" and check for caller number validity
399     String[] caller = args[i+1].split("-");
400     checkInt(caller, "caller");
401     checkNumberFormat(args[i+1], "caller");
402 
403     // Split with "-" and check for callee number validity
404     String[] callee = args[i+2].split("-");
405     checkInt(callee, "callee");
406     checkNumberFormat(args[i+2], "callee");
407 
408     // Split with "/" and check for starting date validity
409     String[] start = args[i+3].split("/");
410     checkInt(start, "starting date");
411     checkDate(start, args[i+3], "starting date's");
412 
413     // Split with ":" and check for starting time validity
414     String[] startTime = args[i+4].split(":");
415     checkInt(startTime, "starting time");
416     checkTime(startTime, args[i+4], "starting time");
417 
418     // Split with "/" and check for ending date validity
419     String[] end = args[i+5].split("/");
420     checkInt(end, "ending date");
421     checkDate(end, args[i+5], "ending date's");
422 
423     // Split with ":" and check for ending time validity
424     String[] endTime = args[i+6].split(":");
425     checkInt(endTime, "ending time");
426     checkTime(endTime, args[i+6], "ending time");
427 
428     // Initialize PhoneCall class with verified command line input
429     calling.initCall(caller, callee, start, end, startTime, endTime);
430   }

493   /**
494    * Check if the command line fed PhoneCall information already exists in the PhoneBill class
495    *
496    * @param calls
497    *         Collection of PhoneCall in PhoneBill
498    *
499    * @param calling
500    *         PhoneCall initialized from command line argument
501    */
502   public static void checkCallDupli(Collection<PhoneCall> calls, PhoneCall calling){
503     Iterator<PhoneCall> iter = calls.iterator();      // Set up Iterator to go through all the PhoneCalls
504 
505     // Keep going until no object is left
506     while(iter.hasNext()){
507       PhoneCall obj = iter.next();
508 
509       // If current iteration of PhoneCall is exactly same as PhoneCall from command line, then send out error
510       if(obj.getCaller().equals(calling.getCaller()) && obj.getCallee().equals(calling.getCallee()) && obj.getStartTimeString().equals(calling.getStartTimeString())){
511         if(obj.getEndTimeString().equals(calling.getEndTimeString())){
512           System.err.println("Inputted Phone Call already exists");
513           System.exit(1);
514         }
515       }
516     }
517   }
