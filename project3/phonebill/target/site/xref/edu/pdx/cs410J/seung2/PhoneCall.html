5   /**
6    * PhoneCall class that stores and initialize customer's call information
7    */
8   public class PhoneCall extends AbstractPhoneCall {
9     private String caller = "not implemented";          // Caller's number
10    private String callee = "not implemented";          // Callee's number
11    private String startTime = "not implemented";       // Phone call's starting date and time
12    private String endTime = "not implemented";         // Phone call's ending date and time
13  
14    /**
15     * Initializes the PhoneCall's field with checked but fully parsed fields
16     *
17     * @param Caller
18     *         String array of caller's number in three sections
19     * @param Callee
20     *         String array of callee's number in three sections
21     * @param sDate
22     *         String array of starting date which has month, date, and year value
23     * @param eDate
24     *         String array of ending date which has month, date, and year value
25     * @param sTime
26     *         String array of starting time which has hour and min
27     * @param eTime
28     *         String array of ending time which has hour and min
29     */
30    public void initCall(String[] Caller, String[] Callee, String[] sDate, String[] eDate, String[] sTime, String[] eTime){
31      // Initialize caller and callee's phone number with correct format
32      this.caller = Caller[0] + "-" + Caller[1] + "-" + Caller[2];
33      this.callee = Callee[0] + "-" + Callee[1] + "-" + Callee[2];
34  
35      // Initialize starting and ending date and time with correct format
36      this.startTime = sDate[0] + "/" + sDate[1] + "/" + sDate[2] + " " + sTime[0] + ":" + sTime[1];
37      this.endTime = eDate[0] + "/" + eDate[1] + "/" + eDate[2] + " " + eTime[0] + ":" + eTime[1];
38    }
39  
40    /**
41     * Returns the caller's number
42     *
43     * @return

45     */

47    public String getCaller() {
48      return this.caller;
49    }

60    }
61

63     * Returns the starting date and time
64     *
65     * @return

67     */
68    @Override
69    public String getStartTimeString() {

71    }

79    @Override

81      return this.endTime;
82    }
